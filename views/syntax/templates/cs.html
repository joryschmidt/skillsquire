class Person
{
    // Fields
    public readonly string FirstName;
    public readonly string LastName;
    private string _phonenumber;
    
    // Constructor
    Person(string first, string last, int phone)
    {
        FirstName = first;
        LastName = last;
        _phoneNumber = phone;
    }
    
    // Properties
    public string FullName => FirstName + " " + LastName;
    
    public int PhoneNumber
    {
        get
        {
            return _phoneNumber;
        }
        set
        {
            _phoneNumber = value;
        }
    }
    
    OR EQUIVALENTLY
