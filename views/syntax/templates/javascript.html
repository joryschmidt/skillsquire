Javascript


Hello World

console.log("hello world");


Function Definition

function helloWorld() {

    console.log("hello world");

}

OR

var helloWorld = function() {

    console.log("hello world");

}


For Loop

for (var i = 0; i < 10; i++) {

    console.log(i);

}


While Loop

var count = 0;

while (count < 10) {

    console.log(count);

    count++;

}

OR

do {

    code

} while (count < 10);


Conditionals

if (conditional) {

    // Do something

} else if (conditional) {

    // Do something else

} else {

    // Do default

}


Switch Statement

var food = 'Veggies'

switch (food) {

    case 'Fruit':

        code

        break;

    case 'Veggies':

        code

        break;

    default:

        code

}

&& || !


Strings

"" OR ''

var name = 'world';

console.log("hello " + name);

console.log(`hello ${name}`);
